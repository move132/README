
```
git init 

git remote add origin https://github.com/username/object.git

git add .  or  git add filename

git commit -am '注释' --提交

git branch  --查看分支

git checkout --file --撤回

git checkout XXX 切换分支

git checkout -b 分支名称  --创建并切换到分支

git merge   分支名称   --合并到当前分支

git reset --merge  --撤回合并分支
```
